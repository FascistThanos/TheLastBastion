#Proposal - BAW and BAV get the event

country_event = {
  id = southgermany.1
  title = southgermany.1.t
  desc = southgermany.1.d

  fire_only_once = yes

  trigger = {
    GER = {
      OR = {
        has_war_with = BAV
        has_war_with = BAW
      }
    }
    AND = {
      OR = {
        original_tag = BAW
        original_tag = BAV
      }
    }
  }

  option = {
    name = southgermany.1.a
    BAW = {
      if = {
        limit = {
          is_ai = yes
        }
        country_event = southgermany.2
      }
    }
    BAV = {
      if = {
        limit = {
          is_ai = yes
        }
        country_event = southgermany.2
      }
    }
  }
  option = {
    name = southgermany.1.b
  }
}
