#Soviet-American War
news_event = {
	id = news.1
	title = news.1.t
	desc = news.1.desc

	major = yes

	trigger = {
		has_global_flag = soviet_american_war
	}

	fire_only_once = yes

	option = {
		name = news.1.a
		trigger = {
			NOT = {
				OR = {
				TAG = USA
				TAG = SOV
				is_in_faction_with = USA
				is_in_faction_with = SOV
			}
			}
		}
	}
	option = {
		name = news.1.b
		trigger = {
			OR = {
				TAG = USA
				is_in_faction_with = USA
			}
		}
	}

	option = {
		name = news.1.c
		trigger = {
			OR = {
				TAG = SOV
				is_in_faction_with = SOV
			}
		}
	}
}
